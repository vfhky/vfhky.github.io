#include <stdio.h>
void fun(int *a, int *n)
{
   int i = 2000, sum = 0;
   for(;i<2501;i++)
   {
      if( (i%7==0 || i%17==0 || i%27==0) && (!(i%7==0  && i%17==0)) && (!(i%17==0 && i%27==0)) )
      {
           *a++ = i;
           sum++;
      }
   }
   /* 返回符合要求元素的个数 */
   *n = sum;
}

int main()
{
   int a[120];
   int k;
   fun(a,&k);
   /* 使指针pt指向数组a的起始地址 */
   int *pt = a;
   for(;pt<a+120;pt++)
   {
         printf("%d\n",*pt);
   }
   printf("%d\n",k);
   return 0;
}
